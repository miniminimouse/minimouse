/* Step 1: Get first 20 unique loan numbers */
proc sql;
   create table first_20_loans as
   select distinct loan_number
   from your_dataset_name;
quit;

data first_20_loans;
   set first_20_loans;
   if _n_ <= 20;
run;

/* Step 2: Get all rows for those 20 loans */
proc sql;
   create table subset_data as
   select a.*
   from your_dataset_name a
   inner join first_20_loans b
   on a.loan_number = b.loan_number;
quit;
