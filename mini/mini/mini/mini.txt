def paginate_df(df, page_size=500):
    total_pages = len(df) // page_size + 1
    
    def show_page(page=0):
        start_idx = page * page_size
        end_idx = min((page + 1) * page_size, len(df))
        print(f"Page {page+1} of {total_pages} (rows {start_idx} to {end_idx-1})")
        display(df.iloc[start_idx:end_idx])
    
    return show_page
